


   ```python
   for category in ["csd", "danskvand", "energidrikke", "rtd", "totalbeer", "pooled_4", "pooled_5"]:
       fm_path = ANALYSIS_DIR / "outputs" / category / "feature_matrix_split.parquet"
   ```
